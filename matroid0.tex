    \subsection{集合に関して}
      \plainpar{}{
        この文書で扱う集合はおおかた有限集合である．
        したがって，集合について必要以上に公理的に見ることはせず，比較的素朴に扱う方針をとる．
        $X$ が $Y$ の部分集合であることを $X \subseteq Y$ と書き，しばしば $Y$ が $X$ を包むと言い表す．
        $X$ の部分集合全体すなわち $X$ の冪集合は $\Power{X}$ と書く．
        $X \subseteq Y$ かつ $X \neq Y$ であることを $X \subsetneq Y$ と書き，しばしば $Y$ が $X$ を真に包むと言い表す．
        $X$ と $Y$ の和集合は $X \cup Y$，$X$ と $Y$ の直積は $X \times Y$ と書く．
        なお，集合 $S$ と述語 $P$ を用いた集合の内包記法 $\setprnsep{x \in S}{\app{P}{x}}$ は分出公理を意識したものである．
        しばしば左側にも写像 $f$ を用いて $\setprnsep{\app{f}{x}}{\app{P}{x}}$ のように書くが，
        これは $\setprnsep{y \in \cod f}{\existsin{x}{\dom f}{\lfland{y = \app{f}{x}}{\app{P}{x}}}}$ の略記である．
      }
      \defpar{共通部分}{
        集合 $A$，$B$ に対し，$A \cap B \defeq \setprnsep{x \in A}{x \in B}$ を
        $A$ と $B$ の\newwordjaen{共通部分}{intersection}と呼ぶ%
          \footnote{稀に共通部分のことを積集合と呼ぶことがあるが，直積との混同を避けるため使わないことにする．
          また，和集合も直和との混同を避けてか\newwordjaen{合併}{union}と呼んだりする．}．
      }
      \defpar{差，対称差}{
        集合 $A$，$B$ に対し，
          $A \setmns B \defeq \setprnsep{x \in A}{x \not\in B}$
        を $A$ と $B$ の\newwordjaen{差}{difference}或いは\newword{差集合}と呼ぶ．また，
          $A \symdiff B \defeq \paren{A \setmns B} \cup \paren{B \setmns A}$
        を $A$ と $B$ の\newwordjaen{対称差}{symmetric difference}と呼ぶ．
      }
      \defpar{二項関係}{
        集合 $A$ と $B$ に対し，$R \subseteq A \times B$ を $A$ と $B$ の上の\newwordjaen{二項関係}{binary relation}と呼ぶ．
        特に $R \subseteq A \times A$ を $A$ 上の二項関係と呼ぶ．
        また，$a \in A$ と $b \in B$ に対して二項関係 $R \subseteq A \times B$ が
        $\seqprn{a, b} \in R$ を満たすとき，これを $a \mathrel{R} b$ と中置演算子に見せかけた糖衣構文で書く．
      }
      \defpar{写像}{
        集合 $A$，$B$，二項関係 $f \subseteq A \times B$ が
        \begin{align*}
          \forallin[1]{a}{A}{\existsin{b}{B}{a \mathrel{f} b}}
        \\
          \forallin[1]{a}{A}{\forallin{b_{1}|b_{2}}{B}{
            \lflimpl{\lfland{a \mathrel{f} b_{1}}{a \mathrel{f} b_{2}}}{b_{1} = b_{2}}
          }}
        \end{align*}
        を共に満たすとき，すなわち任意の $a \in A$ に対して $a \mathrel{f} b$ なる $b \in B$ が一意的に存在するとき，
        $f$ を $A$ から $B$ への\newwordjaen{写像}{map}と呼ぶ．
        $A$ を $f$ の\newwordjaen{定義域}{domain of definition}或いは\newwordjaen{始域}{domain}，
        $B$ を $f$ の\newwordjaen{値域}{range}或いは\newwordjaen{終域}{codomain}と呼び，
        $f$ が $A$ から $B$ への写像であることを $\funcdoms{f}{A}{B}$ と書く．
        また，写像 $\funcdoms{f}{A}{B}$ と $a \in A$ に対して一意的に存在する $a \mathrel{f} b$ なる $b \in B$ を
        $\app{f}{a}$ と書く．
      }
      \defpar{写像の制限}{
        写像 $\funcdoms{f}{A}{B}$ と $X \subseteq A$ に対し，
          \funcdomswidebox{\restrfunc{f}{X}}{X}{B}{x}{\app{f}{x}}
        で定義される $\restrfunc{f}{X}$ を $f$ の $X$ への\newword{制限}と呼ぶ．
      }
      \defpar{像，逆像}{
        写像 $\funcdoms{f}{A}{B}$ と集合 $X \subseteq A$ に対して
          $\funcimg{f}{X} \defeq \setprnsep{y \in B}{\existsin{x}{X}{y = \app{f}{x}}}$
        を $f$ による $X$ の\newwordjaen{像}{image}と呼ぶ．また，
        写像 $\funcdoms{f}{A}{B}$ と集合 $Y \subseteq B$ に対して
          $\funcinvimg{f}{Y} \defeq \setprnsep{x \in A}{\app{f}{x} \in Y}$
        を $f$ による $Y$ の\newwordjaen{逆像}{inverse image}と呼ぶ．
      }
      \defpar*{極大・極小集合全体の記法}{
        集合族 $\famF$ に対し，
        \begin{align*}
          \maxl \famF &\defeq \setprnsep{X \in \famF}{\forallin{Y}{\famF}{\lflimpl{Y \supseteq X}{Y = X}}}
        \\
          \minl \famF &\defeq \setprnsep{X \in \famF}{\forallin{Y}{\famF}{\lflimpl{Y \subseteq X}{Y = X}}}
        \end{align*}
        で $\maxl \famF$，$\minl \famF$ を定める．
      }
      \complpar{}{
        同一の性質を持った集合族 $\famF$ の元のうち包含関係に関して極大であるもの全体が $\maxl \famF$，
        極小であるもの全体が $\minl \famF$ である．
      }

    \subsection{グラフに関して}
      \plainpar{}{
        有限個の\newwordjaen{頂点}{vertex}と，
        それらの頂点のうち順序の区別なくまた重複を許して $2$ 個を結ぶ有限本の\newwordjaen{辺}{edge}からなる図形的構造を
        \newwordjaen{無向グラフ}{undirected graph}と呼ぶ．
        “頂点と辺をどのように配置するか”については考慮に入れる場合と入れない場合があるが，
        入れない場合，“代数的には”次のように定義される．
      }
      \defpar{無向グラフ}{
        有限集合 $V$，$E$ と写像 $\funcdoms{\edtovtx}{E}{\Power{V}}$ が
        \begin{align*}
          &\forallin{e}{E}{1 \leq \card{\app{\edtovtx}{e}} \leq 2}
        \end{align*}
        を満たすとき，$G = \seqprn{V, E; \edtovtx}$ を
        \newwordjaen{無向グラフ}{undirected graph}或いは単に\newwordjaen{グラフ}{graph}と呼び，
        $V$ を $G$ の\newwordjaen{頂点集合}{vertex set}，$E$ を $G$ の\newwordjaen{辺集合}{set of edges}と呼ぶ．
        また $e \in E$ に対して $v \in V$ が $v \in \app{\edtovtx}{e}$ を満たすとき，$v$ を $e$ の\newwordjaen{端点}{endpoint}と呼ぶ．
        なお， $\edtovtx$ を意識しなくてよい場合は $G = \seqprn{V, E; \edtovtx}$ を $G = \seqprn{V, E}$ と略記する．
      }
      % ----
      \begin{figure}[htb]
      \centering
        \parbox[c]{0.3\textwidth}{\input{img/undirected-graph.tex}}
        \parbox[c]{0.3\textwidth}{
          \begin{align*}
            V &= \setprn{v_{1}, v_{2}, v_{3}}
          \\
            E &= \setprn{e_{1}, e_{2}, e_{3}, e_{4}, e_{5}}
          \\
            \app{\edtovtx}{e_{1}} &= \setprn{v_{1}, v_{2}}
          \\
            \app{\edtovtx}{e_{2}} &= \app{\edtovtx}{e_{3}} = \setprn{v_{2}, v_{3}}
          \\
            \app{\edtovtx}{e_{4}} &= \setprn{v_{1}, v_{3}}
          \\
            \app{\edtovtx}{e_{5}} &= \setprn{v_{1}}
          \end{align*}
        }
        \caption{グラフの例}\label{fig:undirected-graph}
      \end{figure}

      \defpar{隣接性}{
        グラフ $G = \seqprn{V, E; \edtovtx}$ に於いて $e \in E$ が $\card{\app{\edtovtx}{e}} = 2$ を満たして
        $\app{\edtovtx}{e} = \setprn{u, v}$ であるとき，$e$ は $u$ と $v$ を\newwordjaen{結ぶ}{joint}という．
        また，$u \neq v$ なる $u, v \in V$ に対し，$\app{\edtovtx}{e} = \setprn{u, v}$ なる $e \in E$ が存在するとき，
        $u$ と $v$ は\newwordjaen{隣接している}{adjacent}という．
      }
      \defpar{ループ，並行性}{
        グラフ $G = \seqprn{V, E; \edtovtx}$ に於いて $e \in E$ が $\card{\app{\edtovtx}{e}} = 1$ を満たすとき，
        $e$ を\newwordjaen{ループ}{loop}と呼ぶ．
        また，$e_{1}, e_{2} \in E$ が $\app{\edtovtx}{e_{1}} = \app{\edtovtx}{e_{2}}$ かつ
        $\card{\app{\edtovtx}{e_{1}}} = \card{\app{\edtovtx}{e_{2}}} = 2$ を満たすとき，
        $e_{1}$ と $e_{2}$ は\newwordjaen{並行している}{parallel}という．
      }
      \example{}{
        グラフの例としては図\ref{fig:undirected-graph}のようなものが挙げられる．
        $e_{2}$ と $e_{3}$ は並行しており，$e_{5}$ はループである．
      }
      \defpar{多重グラフ，単純グラフ}{
        グラフ $G = \seqprn{V, E; \edtovtx}$ が $\forallin{e}{E}{\card{\app{\edtovtx}{e}} = 2}$ を満たすとき，
        $G$ を\newwordjaen{多重グラフ}{multigraph}と呼ぶ．
        さらに，多重グラフ $G = \seqprn{V, E; \edtovtx}$ の $\edtovtx$ が単射であるとき，
        $G$ を\newwordjaen{単純グラフ}{simple graph}と呼ぶ．
      }
      \complpar{}{
        要するに多重グラフとはループを持たないグラフ，単純グラフとはループを持たずどの $2$ 辺も並行しないグラフである．
        また，単純グラフでは $\edtovtx$ の単射性より $V$ の $2$ 元部分集合全体 ${V \choose 2}$ を用いて
        $E \subseteq {V \choose 2}$，$\edtovtx = \idem_{E}$ としてよい．
        こうすることで端点に $u$ と $v$ を持つ辺 $e$ は $e = \setprn{u, v}$ と表せる．
        端点の順序は区別しないから，組 $\seqprn{u, v} \in V \times V$ ではなく
        非順序対 $\setprn{u, v} \in {V \choose 2}$ であることに注意されたい．
        勿論 ${V \choose 2}$ から $V \times V$ への単射が存在するから
        $E \subseteq V \times V$ と“広めにとっておく”こともできるが，
        この場合 $u \neq v$ なる $u,v \in V$ に対して
        $\seqprn{u, v} \in E$ ならば $\seqprn{v, u} \not\in E$ でなければならないから，扱いが煩雑になる．
      }
      \defpar{頂点の次数}{
        グラフ $G = \seqprn{V, E; \edtovtx}$ に於いて，$v \in V$ に対して
        \begin{align*}
          \deg_{G} v
%            &\defeq \card{\setprnsep{e \in E}{v \in \app{\edtovtx}{e}}}
%              + \card{\setprnsep{e \in E}{\setprn{v} = \app{\edtovtx}{e}}}
            &\defeq \card{\setprnsep{e \in E}{\lfland{\card{\app{\edtovtx}{e}} = 2}{v \in \app{\edtovtx}{e}}}}
              + 2 \cdot \card{\setprnsep{e \in E}{\lfland{\card{\app{\edtovtx}{e}} = 1}{v \in \app{\edtovtx}{e}}}}
        \end{align*}
        で定義される $\deg_{G} v$ を $v$ の\newwordjaen{次数}{degree}と呼ぶ．
        $\deg_{G} v = 0$ なる $v \in V$ は\newwordjaen{孤立している}{isolated}という．
        また，グラフ $G = \seqprn{V, E; \edtovtx}$ が
        任意の $u, v \in V$ に対して $\deg_{G} u = \deg_{G} v$ を満たすとき，
        すなわちすべての頂点の次数が相等しいとき，$G$ を\newwordjaen{正則グラフ}{regular graph}と呼ぶ．
      }
      \complpar{}{
        特に $G = \seqprn{V, E; \edtovtx}$ が多重グラフのときはループがないので，$v \in V$ の次数は
        \begin{align*}
          \deg_{G} v &= \card{\setprnsep{e \in E}{v \in \app{\edtovtx}{e}}}
        \end{align*}
        である．
      }
      \example{}{
        図\ref{fig:undirected-graph}の各頂点の次数は
        $\deg_{G} v_{1} = 4$，$\deg_{G} v_{2} = 3$，$\deg_{G} v_{3} = 3$．
      }
      \defpar{完全グラフ}{
        $\card{V} = n$ なる単純グラフ $G = \seqprn{V, E; \edtovtx}$ に対して $E \cong {V \choose 2}$ であるとき，
        すなわち任意の $2$ 点に対してそれらを結ぶ辺がただ $1$ つ存在するとき，
        $G$ を\newwordjaen{完全グラフ}{complete graph}と呼び，$K_{n}$ と書く．
      }
      \defpar{安定性}{
        単純グラフ $G = \seqprn{V, E; \edtovtx}$ に於いて $U \subseteq V$ が
        \begin{align*}
          \forallin{u|v}{U}{\forallin{e}{E}{\app{\edtovtx}{e} \neq \setprn{u, v}}}
        \end{align*}
        を満たすとき，すなわち $U$ のどの $2$ 元も隣接しないとき，
        $U$ を $G$ の\newwordjaen{安定集合}{stable set}と呼ぶ%
          \footnote{安定集合を独立集合と呼ぶこともあるが，これはマトロイドに関して使用する全く異なる用語と重複するので避ける．}．
      }
      \defpar{二部グラフ}{
%        \begin{multline*}
%          \existssub{V_{1}}{V}{
%            \lfland{
%              \lfland{V_{1} \neq \varnothing}{V_{1} \neq V}
%            }{\midbreak
%              \forallin{e}{E}{
%                \lfland{
%                  \forallin{u|v}{V_{1}}{\app{\edtovtx}{e} \neq \setprn{u, v}}
%                }{
%                  \forallin{u|v}{V \setmns V_{1}}{\app{\edtovtx}{e} \neq \setprn{u, v}}
%                }
%              }
%            }
%          }
%        \end{multline*}
        単純グラフ $G = \seqprn{V, E; \edtovtx}$ に対し，或る $V_{1} \subseteq V$ が存在して
        $V_{1}$ と $V \setmns V_{1}$ が共に非空な安定集合であるとき，
        $G$ を\newwordjaen{二部グラフ}{bipartite graph}と呼ぶ．
        $G$ が二部グラフで特に上のような $V_{1}$ を明示したいとき，
        $V_{2} \defeq V \setmns V_{1}$ として $G = \Bipartite{V_{1}}{V_{2}}{E; \edtovtx}$，
        或いは $\edtovtx$ を省略して $G = \Bipartite{V_{1}}{V_{2}}{E}$ と書く．
      }
      \complpar{}{
        二部グラフは，或る $V_{1} \subseteq V$ が存在して任意の $e \in E$ に対して
        $\card{\app{\edtovtx}{e} \cap V_{1}} = 1$ が成り立つような単純グラフでもある．
        上の定義との同値性は直感的にも明らかであり，こちらを定義としてもよい．
      }
      \defpar{完全二部グラフ}{
        $\card{V_{1}} = n_{1}$，$\card{V_{2}} = n_{2}$ なる二部グラフ $G = \Bipartite{V_{1}}{V_{2}}{E; \edtovtx}$ が
        任意の $v_{1} \in V_{1}$ に対して $\deg_{G} v_{1} = n_{2}$ かつ
        任意の $v_{2} \in V_{2}$ に対して $\deg_{G} v_{2} = n_{1}$ を満たすとき，
        $G$ を\newwordjaen{完全二部グラフ}{complete bipartite graph}と呼び，$K_{n_{1}, n_{2}}$ と書く．
      }
      \defpar{道}{
        グラフ $G = \seqprn{V, E; \edtovtx}$ と $n \in \setN$ に対し，
        頂点の列と辺の列の組
        $P = \seqprn{\seqprn{v_{i}}_{i = 0}^{n}, \seqprn{e_{i}}_{i = 1}^{n}} \in V^{n + 1} \times E^{n}$ が
        \begin{align*}
          \forallin[1]{i}{\Natleq{n}}{\app{\edtovtx}{e_{i}} = \setprn{v_{i - 1}, v_{i}}}
        \\
          \forallin[1]{i|j}{\Natleq{n}}{\lflimpl{i \neq j}{e_{i} \neq e_{j}}}
        \end{align*}
        を満たすとき，
        $P$ を $v_{0}$ と $v_{n}$ を\newwordjaen{繋ぐ}{connect}\newwordjaen{長さ}{length} $n$ の\newwordjaen{道}{path}と呼ぶ．
        また，このとき $v_{0}$ を $P$ の\newwordjaen{始点}{initial vertex}，
        $v_{n}$ を $P$ の\newwordjaen{終点}{terminal ventex}と呼ぶ．
      }
      \complpar{}{
        定義より，道には同一の辺が複数回現れることはないが，頂点は重複してもよい．また，始点と終点は区別する．
        その他，特に $v \in V$ に対して $v$ と $v$ を繋ぐ長さ $0$ の道 $\seqprn{\seqprn{v}, \seqprn{}}$ も
        定義されていることに注意されたい．
      }
      \defpar{連結性，連結成分}{
        グラフ $G = \seqprn{V, E}$ に対し，$V$ 上の二項関係 $\connectrel$ を，
        $u$ と $v$ を繋ぐ道が $G$ に存在するときに $u \connectrel v$ が成り立つものと定義する．
        この $\connectrel$ は明らかに同値関係であり，
        $V$ の $\connectrel$ による同値類の各元を $G$ の\newwordjaen{連結成分}{connected components}と呼ぶ．
        また，任意の $u, v \in V$ に対して $u \connectrel v$ が成り立つとき，
        $G$ は\newwordjaen{連結である}{connected}という．
        連結でないことは\newwordjaen{非連結である}{disconnected}という．
      }
      \defpar{部分グラフ，全域部分グラフ，部分グラフの生成}{
        グラフ $G = \seqprn{V, E; \edtovtx}$ と $V' \subseteq V$ と $E' \subseteq E$ に対して
        $G' = \seqprn{V', E'; \restrfunc{\edtovtx}{E'}}$ がグラフをなすとき，
        $G'$ を $G$ の\newwordjaen{部分グラフ}{subgraph}と呼ぶ．
        $\edtovtx$ を意識しなくてよい場合は，単に $\seqprn{V', E'}$ を $G$ の部分グラフと呼ぶ．
        また，辺集合がもとのグラフ $G$ の辺集合 $V$ と一致する部分グラフを
        $G$ の\newwordjaen{全域部分グラフ}{spanning subgraph}と呼ぶ．
        さらに，辺部分集合 $E' \subseteq E$ に対して
        \begin{align*}
          \Vertexgenby{E'} \defeq \setprnsep{v \in V}{\existsin{e}{E'}{v \in \app{\edtovtx}{e}}}
        \end{align*}
        で頂点部分集合が定義される部分グラフ $\seqprn{\Vertexgenby{E'}, E'}$ を，
        $E'$ が\newwordjaen{生成する}{generate} $G$ の部分グラフと呼び，
        頂点部分集合 $V' \subseteq V$ に対して
        $G$ から $V'$ の元を端点にもつ辺を除いた，頂点集合を $V'$ とする部分グラフを $G \rmvtx V'$ と書く．
        すなわち
        \begin{align*}
          G \rmvtx V' \defeq \seqprn{V', \setprnsep{e \in E}{\forallin{v}{V'}{v \not\in \app{\edtovtx}{e}}}}
        \end{align*}
        である．
      }
      \defpar{多重連結性，関節点}{
        グラフ $G = \seqprn{V, E}$ と $n \in \setNpos$ に対し，
        任意の $\card{U} < n$ なる $U \subseteq V$ に対して $G \rmvtx U$ が連結であるとき，
        $G$ は $n$\newwordjaen{-連結}{-connected}であるという．
        また，連結なグラフ $G = \seqprn{V, E}$ に対し，$v \in V$ が $G \rmvtx \setprn{v}$ を非連結にするとき，
        $v$ を $G$ の\newwordjaen{関節点}{cut vertex}と呼ぶ．
      }
      \defpar{閉路}{
        グラフ $G = \seqprn{V, E; \edtovtx}$ と $n \in \setNpos$ に対し，
        $C = \seqprn{\seqprn{v_{i}}_{i = 0}^{n - 1}, \seqprn{e_{i}}_{i = 0}^{n - 1}} \in V^{n} \times E^{n}$ が
        \begin{align*}
          \forallin[1]{i|j}{\Natintvl{0}{n - 1}}{\lflimpl{i \neq j}{v_{i} \neq v_{j}}}
        \\
          \forallin[1]{i|j}{\Natintvl{0}{n - 1}}{\lflimpl{i \neq j}{e_{i} \neq e_{j}}}
        \\
          \forallin[1]{i}{\Natintvl{0}{n - 1}}{\app{\edtovtx}{e_{i}} = \setprn{v_{i}, v_{(i + 1) \bmod n}}}
        \end{align*}
        を満たすとき，$C$ を $G$ の\newwordjaen{閉路}{cycle}と呼ぶ．
        或いは，$C$ からつくられる $G$ の部分グラフ
          $\seqprn{\setprn{v_{i}}_{i = 0}^{n - 1}, \setprn{e_{i}}_{i = 0}^{n - 1}}$
        も単に閉路と呼ぶ．
      }
      \complpar{}{
        道と異なり，閉路では辺だけでなく頂点も重複を許容しない．
      }

      }
